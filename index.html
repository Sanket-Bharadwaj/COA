Arrange the elements of array, ascending order.txt

.DATA
count db 06
value db 09H, OFH, 14H, 45H, 24H,
.CODE
MAIN PROC
MOV AX, DATA
MOV DS, AX
LEA DI, count
MOV CH, [DI]
DEC CH
UP2: MOV CL, CH
LEA SI, value
UP1: MOV AL, [SI]
CMP AL, [Si+1]
JC DOWN
MOV DL, [SI+1]
XCHG [SI], DL
MOV [SI+1], DL

DOWN: INC SI
DEC CL
JNZ UP1
DEC CH
JNZ UP2
END MAIN

Find the largest number in a given array of size N.txt

.data
count db 04h
value db 09h, 10h, 05h, 03h
res db ?
.code
MAIN PROC
mov ax, data
mov ds, ax
mov cl, count
dec cl
LEA SI, value
mov al, [si]
up: inc si
cmp al, [si]
jnc nxt
mov al, [si]
nxt: dec cl
jnz up
LEA di, res
mov [di], al
end main
